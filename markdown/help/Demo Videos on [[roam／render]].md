(defn text-input [x]
   [:div 	
   [:input {:value @x
              :on-change 
      (fn [e] 
         (reset! x (.. e -target -value)))}]
             ]
          )
```
                    - ```clojure
(defn query-off-string [x]
  (let [y (dr/q '[:find ?x ?s
                  :in $ ?title
                  :where 
                  [?e :node/title ?title]
                  [?x :block/refs ?e]
                  [?x :block/string ?s]
                  ]
                @x
                )]
    [:div (pr-str (take 10 @y))]
  ))```
